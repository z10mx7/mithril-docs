# Select Component

The `Select` component provides an intuitive dropdown menu that supports both single and multiple selections with an optional search input.

import { useState } from 'react';
import {
  Select,
  SelectValue,
  SelectContent,
  SelectItem,
} from '@/components/select';

export const SelectSingle = () => {
  const [selectedFruits, setSelectedFruit] = useState('apple');

  const handleSelect = (value: string | string[]) => {
    setSelectedFruit(value as string);
  };

  return (
    <Select
      search

      className="w-70"

## Props

### `<Select>`

| Prop        | Type            | Default            | Description                         |
| ----------- | --------------- | ------------------ | ----------------------------------- |
| `value`     | `string         | string[]`          | `null`                              |
| `onSelect`  | `(value: string | string[]) => void` | `-`                                 |
| `multiple`  | `boolean`       | `false`            | Enables multiple selections.        |
| `search`    | `boolean`       | `false`            | Enables search within the dropdown. |
| `disabled`  | `boolean`       | `false`            | Disables the dropdown.              |
| `className` | `string`        | `-`                | Additional CSS classes for styling. |

### `<SelectValue>`

| Prop          | Type     | Default     | Description                                 |
| ------------- | -------- | ----------- | ------------------------------------------- |
| `placeholder` | `string` | `Select...` | Placeholder text when no value is selected. |
| `className`   | `string` | `-`         | Additional CSS classes.                     |

### `<SelectContent>`

| Prop        | Type     | Default | Description                         |
| ----------- | -------- | ------- | ----------------------------------- |
| `className` | `string` | `-`     | Additional CSS classes for styling. |

### `<SelectItem>`

| Prop        | Type     | Default | Description               |
| ----------- | -------- | ------- | ------------------------- |
| `value`     | `string` | `-`     | Value of the select item. |
| `className` | `string` | `-`     | Additional CSS classes.   |

## Usage Steps

1. Import the `Select` component.
2. Manage the selected state using `value` and `onSelect`.
3. Use `multiple` to enable multiple selections.
4. Enable `search` to allow filtering within the dropdown.

### Example

```tsx
<Select search value={'apple'} onSelect={(value) => console.log(value)}>
  <SelectValue placeholder="Select a fruit" />
  <SelectContent>
    <SelectItem value="apple">Apple</SelectItem>
    <SelectItem value="banana">Banana</SelectItem>
  </SelectContent>
</Select>
```
